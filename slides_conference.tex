%	THIS PAPER (question, approach, findings preview in one)

\begin{frame}{This paper}
\svs
\lead{Research question}
\bit
  \item Does the \hibl{leveraged positioning} of hedge funds amplify sovereign bond yield sensitivity to \hibl{shocks}, and under what conditions?
\eit
\svs
\lead{Approach}
\bit\itemsep2pt
  \item Daily hedge fund positions in \hibl{individual sovereign bonds}, constructed from regulatory repo data
  \item High frequency \hibl{monetary policy surprises} as exogenous shocks {\scriptsize\color{softgrey}\citep{altavilla2019}}
  \item Interact \hibl{predetermined} positions with the shock to recover the additional yield sensitivity that arises when hedge funds trade a given bond
\eit
\svs
\lead{Four findings}
\ben\itemsep2pt
  \item Hedge funds increase the volatility of bond yields \hibl{on shock days}
  \item Their trading \hird{raises} the sensitivity of yields to shocks, a temporary overshooting
  \item The effect is \hird{symmetric}, whether the shock helps or hurts the position
  \item It is concentrated in \hird{directional} books
\een
\end{frame}

\begin{frame}{Specification and identification}
\hypertarget{specification}{}
\svs
\lead{Baseline panel regression}
\beq
\Delta y_{i,t} = \beta_1\,(\text{HF}_{i,t-1}\times \text{Shock}_{t})
 + \beta_2\,\text{HF}_{i,t-1} + \beta_3\,\text{Shock}_{t} + \gamma\,\mathbf{X}_{i,t-1} + \theta_{k} + \varepsilon_{i,t}
\eeq
\svs
\begin{columns}[t]
\begin{column}{0.46\textwidth}
\lead{Positioning and $\beta_1$}
\bit\itemsep3pt
  \item \hibl{Presence}, a binary indicator of an active position, and \hibl{intensity}, the five day net position scaled by amount outstanding
  \item $\beta_1$ is the additional yield sensitivity to the shock when hedge funds are positioned
\eit
\end{column}
\begin{column}{0.50\textwidth}
\lead{Two challenges to a causal reading}
\bit\itemsep3pt
  \item Hedge funds \hird{select} liquid, large, longer duration, and cheapest to deliver bonds
  \item Duration $\times$ country $\times$ date fixed effects compare similar bonds from the same issuer on the same day, and a matched design pairs each bond with its closest twin
  \item A dealer level panel shows the trigger is \hird{not} dealer behavior
\eit
\end{column}
\end{columns}
\vfill
\takeaway{The bond types hedge funds select no longer drive the comparison}
\navbtnTwo{selection}{Selection evidence}{exceedance}{Why these shocks}
\end{frame}

\begin{frame}{Conclusion}
\hypertarget{conclusion}{}
\lvs
\lead{What the paper shows}
\ben\itemsep4pt
  \item Hedge funds select more volatile bonds, but the excess volatility is limited to shock days
  \item Their trading raises the sensitivity of yields to shocks, more than a quarter of the shock day move, and the overshoot reverts within days
  \item The amplification is symmetric across the constraining and relaxing states, consistent with constant leverage targeting
  \item It is concentrated in \hird{directional} books and close to zero in \hibl{hedged} books
\een
\lvs
\lead{For risk monitoring}
\bit\itemsep3pt
  \item Size and presence alone are not sufficient, and amplification is not confined to stress episodes
  \item The \hird{directionality} of the book is a key risk metric beyond position size
\eit

  \item No functional form imposed on how characteristics map into sensitivity, each bond is benchmarked only against its closest twin

\begin{frame}{A numerical example}
\hypertarget{numexample}{}
\setlength\abovedisplayskip{4pt}\setlength\belowdisplayskip{4pt}
\svs
\lead{Setup \quad {\normalfont\color{softgrey} representative sample values}}
\bit\footnotesize\itemsep1pt
  \item A bond of duration about 7 years financed via repo, effective margin $m_i = 2\%$ {\scriptsize(raw haircuts 0.24\%)}
  \item Hedge funds hold $Q_{HF}/Q_{Total} = 3\%$ of the bond
  \item Tightening surprise of 6 bps, market elasticity $\lambda = 5$ bps per pp traded {\scriptsize\color{softgrey}\citep{damico2013, eser2016, desantis2020}}
\eit
\smallskip
\lead{Step 1 \quad {\normalfont the shock erodes the capital buffer}}
\beq
\frac{\Delta E}{E} \approx \frac{1}{m_i}\times\frac{\Delta P}{P}
= \frac{-0.42\%}{2\%} = \hird{-21\%}
\eeq
\lead{Step 2 \quad {\normalfont restoring leverage generates trading pressure}}
\beq
21\% \times 3\% = 0.63\% \text{ of the amount outstanding}
\eeq
\lead{Step 3 \quad {\normalfont trading pressure moves the yield}}
\beq
\Delta y^{\text{Amp}} = \lambda \times 0.63 = 5 \times 0.63 = \hird{\mathbf{3.15 \text{ bps}}}
\eeq

% --- Directionality measure (linked from the Finding 4 slide) ---
\begin{frame}{Book directionality, not only size}
\hypertarget{dirmeasure}{}
\lvs
\lead{The constraint binds at the level of the whole book}
\bit
  \item A book long Italy and short Germany of similar duration carries little net exposure to a parallel rate move, so a shock barely moves its capital
  \item A one sided directional book absorbs the full loss and rebalances
\eit
\lvs
\lead{A bond level measure of holder directionality}
\beq
\text{Dir}_{f,t}=\frac{\left|\sum_i D_i\,q_{f,i,t}\right|}{\sum_i D_i\,|q_{f,i,t}|},
\qquad
\text{HolderDir}_{i,t}=\frac{\sum_f |q_{f,i,t}|\,\text{Dir}_{f,t}}{\sum_f |q_{f,i,t}|}
\eeq
\svs
{\footnotesize Zero when long and short legs offset in duration, close to one when the book is one sided. Each bond inherits the position weighted directionality of the funds holding it.}
\takeaway{A more hedged book behaves like a higher effective margin in the mechanism}

\end{frame}
